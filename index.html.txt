        // --- WhatsApp Logic ---

            window.open(url, '_blank');
        }

        // --- Smooth Scroll ---
        function smoothScroll(e, targetId) {
            e.preventDefault();
            const element = document.getElementById(targetId);
            if (element) {
                element.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
            }
        }

        // --- Mobile Menu ---
        const btnMenu = document.getElementById('mobile-menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');

        btnMenu.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        function closeMobileMenu() {
            mobileMenu.classList.add('hidden');
        }

        // --- Deep Space Canvas Animation ---
        const canvas = document.getElementById('space-canvas');
        const ctx = canvas.getContext('2d');
        const tooltip = document.getElementById('planet-tooltip');

        let width, height;
        let stars = [];
        let planets = [];
        let asteroids = [];
        
        // Mouse Interaction
        let mouseX = 0;
        let mouseY = 0;

        // Assets
        const colors = {
            purple: 'rgba(85, 34, 138, 0.8)',
            // Changed green rgba to orange rgba (230, 122, 46)
            green: 'rgba(230, 122, 46, 0.8)', 
            orange: 'rgba(230, 122, 46, 0.8)',
            white: 'rgba(255, 255, 255, 0.9)'
        };

        // Resize Handling
        function resize() {
            width = window.innerWidth;
            height = window.innerHeight;
            canvas.width = width;
            canvas.height = height;
            initElements();
        }

        // Classes
        class Star {
            constructor() {
                this.x = Math.random() * width;
                this.y = Math.random() * height;
                this.size = Math.random() * 2;
                this.baseSize = this.size;
                this.speedX = (Math.random() - 0.5) * 0.05;
                this.speedY = (Math.random() - 0.5) * 0.05;
                this.brightness = Math.random();
            }

            update() {
                // Parallax based on mouse
                this.x += this.speedX + (mouseX - width/2) * 0.0001 * this.size;
                this.y += this.speedY + (mouseY - height/2) * 0.0001 * this.size;

                // Wrap around
                if (this.x < 0) this.x = width;
                if (this.x > width) this.x = 0;
                if (this.y < 0) this.y = height;
                if (this.y > height) this.y = 0;

                // Twinkle interaction
                const dist = Math.hypot(this.x - mouseX, this.y - mouseY);
                if (dist < 100) {
                    this.size = Math.min(this.baseSize * 2, 4);
                } else {
                    this.size = this.baseSize;
                }
            }

            draw() {
                ctx.fillStyle = `rgba(255, 255, 255, ${this.brightness})`;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
            }
        }

        class Planet {
            constructor(x, y, radius, color, label) {
                this.x = x;
                this.y = y;
                this.radius = radius;
                this.baseRadius = radius;
                this.color = color;
                this.label = label;
                this.hovered = false;
            }

            update() {
                const dist = Math.hypot(this.x - mouseX, this.y - mouseY);
                if (dist < this.radius) {
                    this.radius = this.baseRadius * 1.2;
                    this.hovered = true;
                    document.body.style.cursor = 'pointer';
                } else {
                    this.radius = this.baseRadius;
                    if(this.hovered) { // just left hover
                        this.hovered = false;
                        document.body.style.cursor = 'default';
                        tooltip.style.opacity = 0;
                    }
                }
            }

            draw() {
                // Glow effect
                const gradient = ctx.createRadialGradient(this.x, this.y, this.radius * 0.2, this.x, this.y, this.radius);
                gradient.addColorStop(0, this.color);
                gradient.addColorStop(1, 'transparent');
                
                ctx.fillStyle = gradient;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                ctx.fill();

                // Core
                ctx.fillStyle = this.color.replace('0.8', '1');
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius * 0.3, 0, Math.PI * 2);
                ctx.fill();
            }

            showTooltip() {
                if(this.hovered) {
                    tooltip.innerText = this.label;
                    tooltip.style.left = this.x + 'px';
                    tooltip.style.top = this.y + 'px';
                    tooltip.style.opacity = 1;
                }
            }
        }

        class Asteroid {
            constructor() {
                this.reset();
            }

            reset() {
                this.x = Math.random() < 0.5 ? -20 : width + 20;
                this.y = Math.random() * height;
                this.vx = (Math.random() * 1 + 0.2) * (this.x < 0 ? 1 : -1);
                this.vy = (Math.random() - 0.5) * 1;
                this.size = Math.random() * 3 + 1;
                this.accelerated = false;
            }

            update() {
                this.x += this.vx;
                this.y += this.vy;

                if (this.x < -50 || this.x > width + 50 || this.y < -50 || this.y > height + 50) {
                    this.reset();
                }
            }

            accelerate() {
                const dist = Math.hypot(this.x - mouseX, this.y - mouseY);
                if (dist < 30) {
                    this.vx *= 5;
                    this.vy *= 5;
                    this.accelerated = true;
                }
            }

            draw() {
                ctx.fillStyle = '#888';
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();

                if(this.accelerated) {
                    ctx.strokeStyle = 'white';
                    ctx.beginPath();
                    ctx.moveTo(this.x, this.y);
                    ctx.lineTo(this.x - this.vx * 5, this.y - this.vy * 5);
                    ctx.stroke();
                }
            }
        }

        // Init Elements
        function initElements() {
            stars = [];
            planets = [];
            asteroids = [];

            // Add Stars
            const starCount = width < 768 ? 50 : 150;
            for(let i=0; i<starCount; i++) stars.push(new Star());

            // Add Planets (Background blobs)
            // Top Right
            planets.push(new Planet(width * 0.8, height * 0.2, 80, colors.purple, "Autoridade"));
            // Bottom Left
            planets.push(new Planet(width * 0.1, height * 0.8, 60, colors.green, "Estratégia"));
            // Middle rightish
            if (width > 768) planets.push(new Planet(width * 0.9, height * 0.6, 40, colors.orange, "Impacto"));

            // Add Asteroids
            for(let i=0; i<5; i++) asteroids.push(new Asteroid());
        }

        // Animation Loop
        function animate() {
            // Clear canvas with trail effect
            ctx.fillStyle = 'rgba(10, 10, 10, 0.2)'; // trail effect
            ctx.fillRect(0, 0, width, height);

            // Draw Nebulas (Static background gradients)
            const g = ctx.createRadialGradient(width/2, height/2, 0, width/2, height/2, width);
            g.addColorStop(0, 'rgba(85, 34, 138, 0.05)');
            g.addColorStop(1, 'transparent');
            ctx.fillStyle = g;
            ctx.fillRect(0,0,width,height);

            stars.forEach(star => {
                star.update();
                star.draw();
            });

            planets.forEach(planet => {
                planet.update();
                planet.draw();
            });

            asteroids.forEach(asteroid => {
                asteroid.update();
                asteroid.draw();
            });

            requestAnimationFrame(animate);
        }

        // Event Listeners
        window.addEventListener('resize', resize);
        window.addEventListener('mousemove', (e) => {
            mouseX = e.clientX;
            mouseY = e.clientY;
        });

        window.addEventListener('click', (e) => {
            mouseX = e.clientX;
            mouseY = e.clientY;

            // Click Planets
            planets.forEach(p => p.showTooltip());
            
            // Click Asteroids
            asteroids.forEach(a => a.accelerate());
        });

        // Initialize
        resize();
        animate();

    </script>
